/--
中国式剰余定理（二つのイデアル版）
I と J が coprime → R/(I∩J) ≃+* (R/I) × (R/J)

def chinese_remainder_theorem {I J : Ideal R} (coprime : IsCoprime I J) :
    R ⧸ (I ⊓ J) ≃+* (R ⧸ I) × (R ⧸ J) :=
  Ideal.quotientInfEquivQuotientProd I J coprime

/--
中国式剰余定理（有限個のイデアル版）
互いに coprime な I_i に対して R/(⋂I_i) ≃+* ∏(R/I_i)

def chinese_remainder_general {ι : Type*} [Finite ι] (I : ι → Ideal R) 
    (hI : Pairwise (Function.onFun IsCoprime I)) :
    R ⧸ (⨅ i, I i) ≃+* (i : ι) → (R ⧸ I i) :=
  Ideal.quotientInfRingEquivPiQuotient I hI
