/--
第一同型定理：像-余像同型
任意の射 f: G → H に対して
coim(f) ≃ im(f)

圏論的意義：
- 任意の射の epi-mono 分解
- cokernel-kernel の双対性
- abelian 圏の基本定理
-/
theorem first_isomorphism_categorical 
    {G H : 𝒢} (f : G ⟶ H) [Abelian 𝒢] :
    -- 像と余像の標準同型
    coimage f ≅ image f := by
  -- abelian 圏の基本性質により自動的に成立
  exact Abelian.coimageImageComparison_isIso f

-- ===============================
-- 🔧 第二同型定理の圏論的解釈  
-- ===============================

/--
第二同型定理：格子同型
部分群格子における diamond property

圏論的視点：
- pullback と pushout の双対性
- 格子圏における分配律
- 部分対象分類子の性質
-/
structure DiamondProperty (𝒢 : Type*) [Category 𝒢] where
  -- 対象 G と その部分対象 H, K
  G : 𝒢
  H : Subobject G  
  K : Subobject G
  -- diamond 条件：正規性
  normal_H : Normal H
  -- 結論：格子同型
  diamond_iso : (H ⊔ K) ⧸ H ≅ K ⧸ (H ⊓ K)
