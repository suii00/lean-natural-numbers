section WorkingPatterns

-- ✅ 確実に動作：deriv_mul の基本パターン
example (f g : ℝ → ℝ) (x : ℝ) (hf : DifferentiableAt ℝ f x) (hg : DifferentiableAt ℝ g x) :
    deriv (f * g) x = deriv f x * g x + f x * deriv g x := 
  deriv_mul hf hg

-- ✅ 確実に動作：具体的な関数の積の微分
example (x : ℝ) : deriv (fun t ↦ t^2 * t^3) x = 5 * x^4 := by
  have h1 : DifferentiableAt ℝ (fun t ↦ t^2) x := differentiableAt_pow
  have h2 : DifferentiableAt ℝ (fun t ↦ t^3) x := differentiableAt_pow
  rw [← pow_add] -- t^2 * t^3 = t^5
  simp [deriv_pow]

-- ✅ 確実に動作：段階的な証明アプローチ
example (x : ℝ) : deriv (fun t ↦ t^2 * Real.exp t) x = 2*x * Real.exp x + x^2 * Real.exp x := by
  -- Step 1: 微分可能性の確認
  have h1 : DifferentiableAt ℝ (fun t ↦ t^2) x := differentiableAt_pow
  have h2 : DifferentiableAt ℝ Real.exp x := differentiableAt_exp
  -- Step 2: 積の微分公式を適用
  rw [deriv_mul h1 h2]
  -- Step 3: 個別の導関数を計算
  simp [deriv_pow, deriv_exp]

end WorkingPatterns

claude2.txt では `simp [deriv_add, deriv_const_mul, deriv_pow]` を推奨していましたが、

section ProblematicPatterns

-- ❌ claude2.txt の推奨（失敗する）
example (x : ℝ) : deriv (fun t ↦ t^3 + 2*t^2 + t + 1) x = 3*x^2 + 4*x + 1 := by
  -- simp [deriv_add, deriv_const_mul, deriv_pow, deriv_const]  -- ❌ 失敗
  sorry

-- ✅ 修正版：段階的アプローチ
example (x : ℝ) : deriv (fun t ↦ t^3 + 2*t^2 + t + 1) x = 3*x^2 + 4*x + 1 := by
  -- 個別に処理
  have h1 : deriv (fun t ↦ t^3 + 2*t^2 + t + 1) x = 
           deriv (fun t ↦ t^3) x + deriv (fun t ↦ 2*t^2 + t + 1) x := by
    rw [deriv_add]
    · exact differentiableAt_pow
    · simp [differentiableAt_add, differentiableAt_const_mul, differentiableAt_pow, differentiableAt_const]
  
  have h2 : deriv (fun t ↦ 2*t^2 + t + 1) x = 
           deriv (fun t ↦ 2*t^2) x + deriv (fun t ↦ t + 1) x := by
    rw [deriv_add]
    · simp [differentiableAt_const_mul, differentiableAt_pow]
    · simp [differentiableAt_add, differentiableAt_id, differentiableAt_const]
  
  have h3 : deriv (fun t ↦ t + 1) x = deriv (fun t ↦ t) x + deriv (fun t ↦ 1) x := by
    rw [deriv_add]
    · exact differentiableAt_id
    · exact differentiableAt_const
  
  rw [h1, h2, h3]
  simp [deriv_pow, deriv_const_mul, deriv_id, deriv_const]
  ring

end ProblematicPatterns

section ReliablePatterns

-- パターン1: 基本関数の組み合わせ（個別処理）
example (x : ℝ) : deriv (fun t ↦ t^2 + 3*t) x = 2*x + 3 := by
  rw [deriv_add]
  · exact differentiableAt_pow
  · exact DifferentiableAt.const_mul differentiableAt_id 3
  simp [deriv_pow, deriv_const_mul, deriv_id]

-- パターン2: 指数関数との積
example (x : ℝ) : deriv (fun t ↦ t * Real.exp t) x = Real.exp x + x * Real.exp x := by
  have h1 : DifferentiableAt ℝ (fun t ↦ t) x := differentiableAt_id
  have h2 : DifferentiableAt ℝ Real.exp x := differentiableAt_exp
  rw [deriv_mul h1 h2]
  simp [deriv_id, deriv_exp]

-- パターン3: 三角関数との積
example (x : ℝ) : deriv (fun t ↦ t^2 * Real.sin t) x = 2*x * Real.sin x + x^2 * Real.cos x := by
  have h1 : DifferentiableAt ℝ (fun t ↦ t^2) x := differentiableAt_pow
  have h2 : DifferentiableAt ℝ Real.sin x := differentiableAt_sin
  rw [deriv_mul h1 h2]
  simp [deriv_pow, deriv_sin]

-- パターン4: 合成関数（チェーンルール）
example (x : ℝ) : deriv (fun t ↦ Real.sin (t^2)) x = 2*x * Real.cos (x^2) := by
  have h1 : DifferentiableAt ℝ (fun t ↦ t^2) x := differentiableAt_pow
  have h2 : DifferentiableAt ℝ Real.sin (x^2) := differentiableAt_sin
  rw [deriv.comp]
  · simp [deriv_sin, deriv_pow]
  · exact h2
  · exact h1

end ReliablePatterns

/-!

claude2.txt では `simp [hasDerivAt_pow]` を推奨していましたが、

-/

section HasDerivAtCorrectUsage

-- ❌ claude2.txt の推奨（型推論失敗の可能性）
example (x : ℝ) : HasDerivAt (fun t ↦ t^2) (2*x) x := by
  -- simp [hasDerivAt_pow]  -- ❌ 失敗することがある
  sorry

-- ✅ 修正版：明示的な適用
example (x : ℝ) : HasDerivAt (fun t ↦ t^2) (2*x) x := by
  have h : HasDerivAt (fun t : ℝ ↦ t^2) (2*x) x := hasDerivAt_pow 2 x
  exact h

-- ✅ さらに安全：direct application
example (x : ℝ) : HasDerivAt (fun t ↦ t^2) (2*x) x := 
  hasDerivAt_pow 2 x

-- ✅ 一般的な冪の場合
example (n : ℕ) (x : ℝ) : HasDerivAt (fun t ↦ t^n) (n * x^(n-1)) x := by
  cases' n with n
  · simp [hasDerivAt_const]
  · exact hasDerivAt_pow (n + 1) x

end HasDerivAtCorrectUsage

section ErrorAvoidanceStrategies

-- 戦略1: 段階的な証明構築
def safe_deriv_proof (f g : ℝ → ℝ) (x : ℝ) 
  (hf : DifferentiableAt ℝ f x) (hg : DifferentiableAt ℝ g x) :
  deriv (f + g) x = deriv f x + deriv g x := by
  rw [deriv_add hf hg]

-- 戦略2: 微分可能性の事前確認
lemma poly_differentiable (x : ℝ) : DifferentiableAt ℝ (fun t ↦ t^3 + 2*t^2 + t + 1) x := by
  simp [differentiableAt_add, differentiableAt_pow, differentiableAt_const_mul, differentiableAt_const]

-- 戦略3: 補助定理の活用
lemma deriv_quadratic (a b c : ℝ) (x : ℝ) : 
  deriv (fun t ↦ a*t^2 + b*t + c) x = 2*a*x + b := by
  have h1 : DifferentiableAt ℝ (fun t ↦ a*t^2) x := by
    simp [DifferentiableAt.const_mul, differentiableAt_pow]
  have h2 : DifferentiableAt ℝ (fun t ↦ b*t + c) x := by
    simp [differentiableAt_add, DifferentiableAt.const_mul, differentiableAt_id, differentiableAt_const]
  rw [deriv_add h1 h2]
  simp [deriv_const_mul, deriv_pow, deriv_add, deriv_const, deriv_id]
  ring

end ErrorAvoidanceStrategies

section BestPractices

-- 1. 型の明示的指定
example (x : ℝ) : ℝ := deriv (fun t : ℝ ↦ t^2) x

-- 2. 微分可能性の証明を分離
lemma exp_sin_differentiable (x : ℝ) : 
  DifferentiableAt ℝ (fun t ↦ Real.exp t * Real.sin t) x := by
  exact DifferentiableAt.mul differentiableAt_exp differentiableAt_sin

example (x : ℝ) : deriv (fun t ↦ Real.exp t * Real.sin t) x = 
  Real.exp x * Real.sin x + Real.exp x * Real.cos x := by
  rw [deriv_mul differentiableAt_exp differentiableAt_sin]
  simp [deriv_exp, deriv_sin]

-- 3. 複雑な式の段階的分解
example (x : ℝ) : deriv (fun t ↦ (t^2 + 1) * Real.exp t) x = 
  (2*x) * Real.exp x + (x^2 + 1) * Real.exp x := by
  let f : ℝ → ℝ := fun t ↦ t^2 + 1
  let g : ℝ → ℝ := Real.exp
  
  have hf : DifferentiableAt ℝ f x := by
    simp [f, differentiableAt_add, differentiableAt_pow, differentiableAt_const]
  have hg : DifferentiableAt ℝ g x := by
    simp [g, differentiableAt_exp]
  
  show deriv (f * g) x = deriv f x * g x + f x * deriv g x
  rw [deriv_mul hf hg]
  simp [f, g, deriv_add, deriv_pow, deriv_const, deriv_exp]

end BestPractices

section VerifiedAPIReference

-- 基本的な導関数（確実に動作）
example : deriv (fun x : ℝ ↦ x^2) = fun x ↦ 2*x := by
  ext x
  simp [deriv_pow]

example : deriv (fun x : ℝ ↦ Real.exp x) = Real.exp := by
  ext x
  simp [deriv_exp]

example : deriv (fun x : ℝ ↦ Real.sin x) = Real.cos := by
  ext x
  simp [deriv_sin]

-- 演算規則（事前条件付き）
lemma verified_deriv_add (f g : ℝ → ℝ) (x : ℝ) 
  (hf : DifferentiableAt ℝ f x) (hg : DifferentiableAt ℝ g x) :
  deriv (f + g) x = deriv f x + deriv g x :=
  deriv_add hf hg

lemma verified_deriv_mul (f g : ℝ → ℝ) (x : ℝ) 
  (hf : DifferentiableAt ℝ f x) (hg : DifferentiableAt ℝ g x) :
  deriv (f * g) x = deriv f x * g x + f x * deriv g x :=
  deriv_mul hf hg

lemma verified_deriv_comp (f g : ℝ → ℝ) (x : ℝ)
  (hg : DifferentiableAt ℝ g x) (hf : DifferentiableAt ℝ f (g x)) :
  deriv (f ∘ g) x = deriv f (g x) * deriv g x :=
  deriv.comp hf hg

end VerifiedAPIReference
