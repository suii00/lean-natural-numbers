\section*{哲学（ブルバキ流）}
\begin{itemize}[nosep]
  \item 構造は集合の上に与えられる。順序対 $(x,y)$ は射影 $\pr1,\pr2$ で特徴づける（「順序対は射影で読む」）。
  \item 直積位相は「射影を連続にする最弱の位相」。積空間への連続性は射影合成の連続性で検査できる（普遍性）。
  \item 構造保存写像（群準同型など）は合成に関して閉じ、圏論的整理（対象＋射）が自然。
  \item 位相群では群の基本射（積・逆・単位）が連続。代数と位相の両立性を表す。
\end{itemize}

\section{普遍性：射影で読む連続性}
型 $T,X,Y$ を位相空間とする。$h\colon T\to X\Prod Y$ が連続であることと、$\pr1\circ h,\ \pr2\circ h$ がそれぞれ連続であることは同値：
\begin{theorem}[射影での連続性検査] \label{thm:proj-eqv}
  $\\displaystyle \quad \text{$h$ 連続}\ \Longleftrightarrow\ \big(\text{$\pr1\circ h$ 連続}\ \wedge\ \text{$\pr2\circ h$ 連続}\big)$.
\end{theorem}
\noindent 証明スケッチ：$\Rightarrow$ は合成の連続性。$\Leftarrow$ は成分連続 $\Rightarrow$ ペア連続（$\mathrm{prodMk}$）より従う。

\medskip
普遍性の図式（$f\colon T\to X$, $g\colon T\to Y$ に対し $h=\pair{f}{g}$ を与える）：
\[
\begin{tikzcd}
  & T \arrow[ld, "f"'] \arrow[rd, "g"] \arrow[d, dashed, "h=\pair{f}{g}"] & \\
  X & X\Prod Y \arrow[l, "\pr1"] \arrow[r, "\pr2"'] & Y
\end{tikzcd}
\]

Lean 対応（`BourbakiPairsAndMorphisms.lean`）では、同値 \texttt{continuous\_iff\_proj} と片向き構成子 \texttt{continuous\_of\_proj} を提供し、\texttt{by continuity} と相性良く整備している。

\section{`Prod.map` の正規形と連続性}
2 つの写像 $f\colon X\to Z$, $g\colon Y\to W$ から $\mathrm{Prod.map}(f,g)\colon X\Prod Y\to Z\Prod W$ を得る。次が成り立つ：
\begin{lemma}[正規化（合成の押し込み）]
  $\ (\mathrm{Prod.map}(f_2,g_2))\circ(\mathrm{Prod.map}(f_1,g_1))
     = \mathrm{Prod.map}(f_2\circ f_1,\ g_2\circ g_1)$.
\end{lemma}
\begin{lemma}[恒等の除去]
  $\ \mathrm{Prod.map}(\mathrm{id},\mathrm{id})=\mathrm{id}$.
\end{lemma}
\begin{lemma}[三角恒等式（成分版）]
  $\ \pr1\circ \mathrm{Prod.map}(f,g)= f\circ \pr1,\qquad
     \pr2\circ \mathrm{Prod.map}(f,g)= g\circ \pr2.$
\end{lemma}

\noindent 連続性は、普遍性の片向きで一発：
\begin{theorem}[連続性ラッパ]
  $f,g$ が連続なら $\mathrm{Prod.map}(f,g)$ も連続。\\
  （Lean では `@[continuity] continuous_Prod_map` として提供。）
\end{theorem}

\medskip
合成正規化の図式：
\[
\begin{tikzcd}
  X\Prod Y \arrow[r, "\mathrm{Prod.map}(f_1,g_1)"] \arrow[rr, bend left=15, "\mathrm{Prod.map}(f_2\!\circ\! f_1,\ g_2\!\circ\! g_1)"] &
  Z\Prod W \arrow[r, "\mathrm{Prod.map}(f_2,g_2)"] & Z'\Prod W'
\end{tikzcd}
\]

\section{群準同型の合成と位相群の反転}
\paragraph{群準同型の合成} 群準同型 $\varphi\colon G\to H$, $\psi\colon H\to K$ に対し、合成 $\psi\circ\varphi\colon G\to K$ は再び群準同型。Lean では \texttt{groupHomComp} としてラップし、`@[simp]` で点ごとの正規化（`groupHomComp_apply`）を提供する。

\paragraph{位相群と反転の連続性} 位相群 $G,H$ と連続準同型 $f\colon G\to H$ に対し、
\begin{itemize}[nosep]
  \item 事前合成（定義域の反転）：$g\mapsto f(g^{-1})$ は連続（仮定は $G$ が位相群）。
  \item 事後合成（終域の反転）：$g\mapsto (f(g))^{-1}$ は連続（仮定は $H$ が位相群）。
\end{itemize}
いずれも反転写像の連続性と合成の連続性から従う（Lean: `@[continuity]` 補題として実装）。

\medskip
反転の図式（事後合成）：
\[
\begin{tikzcd}
  G \arrow[r, "f"] \arrow[dr, swap, "g\mapsto (f(g))^{-1}"] & H \arrow[d, "\mathrm{inv}"] \\
  & H
\end{tikzcd}
\]

\section*{Lean API 対応表（抜粋）}
\begin{itemize}[nosep]
  \item 普遍性の同値：\texttt{continuous\_iff\_proj}（内部コア：\texttt{\_core}）
  \item 片向き構成子：\texttt{continuous\_of\_proj}；射影抽出：\texttt{continuous\_proj\_fst/snd}
  \item 連続性ラッパ：\texttt{\@[continuity] continuous\_Prod\_map}
  \item 三角恒等式：\texttt{fst\_comp\_pair}, \texttt{snd\_comp\_pair}, \texttt{fst/snd\_comp\_Prod\_map}
  \item `Prod.map` 正規化：\texttt{\@[simp] prod\_map\_comp}, \texttt{prod\_map\_id\_id}
  \item 群準同型の合成：\texttt{groupHomComp}, \texttt{\@[simp] groupHomComp\_apply}
  \item 位相群の反転合成：\texttt{\@[continuity] topological\_group\_hom\_continuous\_precomp\_inv/postcomp\_inv}
\end{itemize}

\section*{終わりに}
順序対・射影・射の関係を、ブルバキ的「構造＋射」の視点で Lean の API に落とし込み、`simp` と `by~continuity` による安定した収束を狙って正規形を整えた。直積の普遍性を中心に据え、成分と対の間の径路を明示することが要点である。
