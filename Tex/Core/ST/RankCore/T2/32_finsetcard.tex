\subsection{Definition}
Fix a type $\alpha$. Consider $X := \mathrm{Finset}(\alpha)$ and index set $\mathbb{N}$.
\begin{definition}[Finite set cardinality rank]
Define $\rank(S) := |S|$ (i.e.\ \texttt{Finset.card}).
Then
\[
  \layer(n) := \{\, S : \mathrm{Finset}(\alpha) \mid |S| \le n \,\}.
\]
\end{definition}

\subsection{Key lemmas (T2)}
\begin{itemize}
\item \textbf{Layer membership:} \texttt{finset\_layer\_iff}.
\item \textbf{Monotonicity:} \texttt{finset\_layer\_mono}.
\item \textbf{Self-membership:} \texttt{finset\_in\_layer\_self}.
\item \textbf{Subset monotonicity of rank:} \texttt{rank\_subset}.
\end{itemize}
