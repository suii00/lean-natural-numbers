-- ===============================
-- 🔧 準備：型調整のための補助定理
-- ===============================

/--
包含された正規部分群の正規性保持
H ◁ G かつ H ≤ K ≤ G ならば H ◁ K
-/
lemma normal_in_intermediate {G : Type*} [Group G] 
    (H K : Subgroup G) [H.Normal] (h : H ≤ K) :
    (H.subgroupOf K).Normal := by
  -- Mathlibの既存定理を使用
  exact Subgroup.normal_subgroupOf_normal h

-- ===============================
-- 🔧 核心写像の構築
-- ===============================

/--
第二同型定理の核心となる自然写像
K → (H ⊔ K) / H

数学的意味：
- K の各元 k を H ⊔ K の元として見る
- その商類 [k] ∈ (H ⊔ K)/H を対応させる
-/
def natural_map {G : Type*} [Group G] (H K : Subgroup G) [H.Normal] :
    K →* (H ⊔ K) ⧸ H.subgroupOf (H ⊔ K) := by
  -- 包含写像と商写像の合成
  refine QuotientGroup.mk.comp ?_
  -- K ↪ H ⊔ K の包含写像
  exact Subgroup.inclusion le_sup_right

-- ===============================
-- 🔧 核の計算
-- ===============================

/--
自然写像の核の特定
ker(K → (H ⊔ K)/H) = H ∩ K

証明の流れ：
- k ∈ ker ⟺ [k] = [1] in (H ⊔ K)/H
- ⟺ k ∈ H.subgroupOf (H ⊔ K) 
- ⟺ k ∈ H ∩ K (k ∈ K は前提)
-/
lemma natural_map_ker {G : Type*} [Group G] (H K : Subgroup G) [H.Normal] :
    (natural_map H K).ker = (H ⊓ K).subgroupOf K := by
  ext k
  simp only [MonoidHom.mem_ker, natural_map]
  rw [QuotientGroup.eq_one_iff]
  simp only [Subgroup.mem_subgroupOf, Subgroup.mem_inf]
  constructor
  · intro h
    exact ⟨h, k.2⟩
  · intro ⟨h_in_H, _⟩
    exact h_in_H

-- ===============================
-- 🔧 写像の全射性
-- ===============================

/--
自然写像の全射性
任意の [(hk)] ∈ (H ⊔ K)/H に対して、適当な k ∈ K が存在

証明戦略：
- H ⊔ K の任意の元は h * k の形 (h ∈ H, k ∈ K)
- [h * k] = [k] in (H ⊔ K)/H (H が正規なので)
- よって k が前像となる
-/
lemma natural_map_surjective {G : Type*} [Group G] (H K : Subgroup G) [H.Normal] :
    Function.Surjective (natural_map H K) := by
  intro q
  -- 商群の元を代表元で表現
  obtain ⟨g, hg⟩ := QuotientGroup.mk_surjective q
  -- g ∈ H ⊔ K なので g = h * k の形に分解
  obtain ⟨h, k, h_mem, k_mem, rfl⟩ := Subgroup.mem_sup.mp g.2
  use ⟨k, k_mem⟩
  simp only [natural_map, ← hg]
  rw [QuotientGroup.mk_mul_of_normal]
  simp only [QuotientGroup.mk_one]

-- ===============================
-- 🎯 第二同型定理：完全証明
-- ===============================

/--
第二同型定理 (Diamond Isomorphism Theorem)

定理：H ◁ G, K ≤ G に対して
(H ⊔ K) / H ≃* K / (H ∩ K)

ブルバキ的意義：
- 格子理論における diamond property の群論的実現
- 正規化と交叉操作の双対性
- 商構造の普遍的性質の美しい応用

証明の核心：
1. 自然写像 K → (H ⊔ K)/H を構成
2. その核が H ∩ K であることを証明
3. 全射性を示す
4. 第一同型定理により同型を得る
-/
theorem second_isomorphism_theorem {G : Type*} [Group G] 
    (H K : Subgroup G) [H.Normal] :
    (H ⊔ K) ⧸ H.subgroupOf (H ⊔ K) ≃* K ⧸ (H ⊓ K).subgroupOf K := by
  -- 第一同型定理の適用
  have h_ker : (natural_map H K).ker = (H ⊓ K).subgroupOf K := natural_map_ker H K
  have h_surj : Function.Surjective (natural_map H K) := natural_map_surjective H K
  
  -- 同型写像の構成
  rw [← h_ker]
  exact QuotientGroup.quotientKerEquivRange (natural_map H K)
