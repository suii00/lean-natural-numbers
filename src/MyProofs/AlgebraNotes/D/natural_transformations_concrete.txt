/--
核函手：準同型から正規部分群への対応
Kernel functor: from homomorphisms to normal subgroups
-/
def kernel_functor {G H : Type*} [Group G] [Group H] (f : G →* H) :
    Subgroup G := f.ker

/--
像函手：準同型から部分群への対応  
Image functor: from homomorphisms to subgroups
-/
def image_functor {G H : Type*} [Group G] [Group H] (f : G →* H) :
    Subgroup H := f.range

/--
商函手：正規部分群から商群への対応
Quotient functor: from normal subgroups to quotient groups
-/
def quotient_functor {G : Type*} [Group G] {N : Subgroup G} [N.Normal] :
    Type* := G ⧸ N
