section FoundationLayer

/-
  基盤層：確実に動作する群論の基礎
-/

-- 群の第一同型定理：コア実装
theorem group_first_isomorphism {G H : Type*} [Group G] [Group H] (φ : G →* H) :
    (G ⧸ MonoidHom.ker φ) ≃* φ.range := by
  exact QuotientGroup.quotientKerEquivRange φ

-- 準同型の核の正規性
theorem ker_normal {G H : Type*} [Group G] [Group H] (φ : G →* H) :
    (MonoidHom.ker φ).Normal := by
  exact MonoidHom.normal_ker φ

-- 単射性と核の関係
theorem injective_iff_trivial_ker {G H : Type*} [Group G] [Group H] (φ : G →* H) :
    Function.Injective φ ↔ MonoidHom.ker φ = ⊥ := by
  exact MonoidHom.ker_eq_bot_iff.symm

end FoundationLayer
