/-- Prop vs Type の使い分け指針 -/
theorem prop_vs_type_guideline :
  ∀ (complexity_level : ℕ),
  complexity_level < 3 → (∃ (use_prop : Prop), use_prop = True) ∧
  complexity_level ≥ 3 → (∃ (consider_type : Type*), True) := by
  intro complexity_level
  cases' Nat.lt_trichotomy complexity_level 3 with h h
  · left
    constructor
    · exact ⟨True, rfl⟩
    · intro h'
      exfalso
      exact Nat.lt_irrefl _ (Nat.lt_trans h h')
  · cases' h with h h
    · left
      rw [h]
      constructor
      · exact ⟨True, rfl⟩
      · intro h'
        norm_num at h'
    · right
      constructor
      · exact h
      · exact ⟨Empty, trivial⟩
