### Step 1: 証明項の実装

```lean
-- Final.leanの該当部分を修正
noncomputable def add_points {K : Type*} [Field K] (E : EllipticCurve K) :
    Point K E → Point K E → Point K E
  | Point.infinity, Q => Q
  | P, Point.infinity => P
  | Point.affine x₁ y₁ h₁, Point.affine x₂ y₂ h₂ =>
    if h_x : x₁ = x₂ then
      if h_y : y₁ = y₂ then
        if h_zero : y₁ = 0 then Point.infinity
        else
          let slope := (3 * x₁^2 + E.a) / (2 * y₁)
          let x₃ := slope^2 - 2 * x₁
          let y₃ := slope * (x₁ - x₃) - y₁
          Point.affine x₃ y₃ (by
            -- 接線の場合の証明
            simp only [pow_two, pow_three]
            field_simp [h_zero]
            ring_nf
            -- y₃² = x₃³ + a*x₃ + b を証明
            rw [← h₁] -- y₁² = x₁³ + a*x₁ + b を使用
            ring
          )
      else Point.infinity
    else
      let slope := (y₂ - y₁) / (x₂ - x₁)
      let x₃ := slope^2 - x₁ - x₂
      let y₃ := slope * (x₁ - x₃) - y₁
      Point.affine x₃ y₃ (by
        -- 2点を通る直線の場合の証明
        simp only [pow_two, pow_three]
        field_simp [h_x]
        ring_nf
        -- y₃² = x₃³ + a*x₃ + b を証明
        rw [← h₁, ← h₂] -- 両方の点の条件を使用
        ring
      )
```

### Step 2: simple_negateの修正

```lean
def simple_negate (E : EllipticCurve ℚ) : Point ℚ E → Point ℚ E
  | Point.infinity => Point.infinity
  | Point.affine x y h => Point.affine x (-y) (by
    simp only [neg_sq]
    exact h
  )
```
