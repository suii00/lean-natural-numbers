  2. この準同型の核が `J/I` であることを示す

- **証明コード:**
```lean
noncomputable def third_isomorphism_theorem {I J : Ideal R} (h : I ≤ J) :
    (R ⧸ I) ⧸ (J.map (Ideal.Quotient.mk I)) ≃+* R ⧸ J := by
  -- 自然な準同型を構成
  let φ : R ⧸ I →+* R ⧸ J := Ideal.Quotient.lift I (Ideal.Quotient.mk J)
    (fun x hx => by
      rw [Ideal.Quotient.eq_zero_iff_mem]
      exact h hx)
  -- 第一同型定理を適用
  have : Function.Surjective φ := by
    intro y
    obtain ⟨x, rfl⟩ := Ideal.Quotient.mk_surjective y
    use Ideal.Quotient.mk I x
    rfl
  -- 核の特徴付け
  have ker_eq : RingHom.ker φ = J.map (Ideal.Quotient.mk I) := by
    ext x
    simp only [RingHom.mem_ker, Ideal.mem_map_iff_of_surjective]
    constructor
    · intro hx
      obtain ⟨y, rfl⟩ := Ideal.Quotient.mk_surjective x
      use y
      constructor
      · simpa using hx
      · rfl
    · intro ⟨y, hy, rfl⟩
      simp [φ, Ideal.Quotient.lift_mk, hy]
  -- 同型を構成
  rw [← ker_eq]
  exact RingHom.quotientKerEquivOfSurjective this
```
