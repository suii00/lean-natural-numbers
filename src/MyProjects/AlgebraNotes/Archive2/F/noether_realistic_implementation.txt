-- Membership characterization (reliable)
lemma map_mem (f : R →+* S) (I : Ideal R) (s : S) :
    s ∈ Ideal.map f I ↔ ∃ r ∈ I, f r = s := Submodule.mem_map

lemma comap_mem (f : R →+* S) (J : Ideal S) (r : R) :
    r ∈ Ideal.comap f J ↔ f r ∈ J := Submodule.mem_comap

end APIVerification

## Part II: Basic Correspondence Construction

section BasicCorrespondence

variable {R : Type*} [CommRing R] (I : Ideal R)

def forward : {J : Ideal R // I ≤ J} → Ideal (R ⧸ I) :=
  fun ⟨J, _⟩ => Ideal.map (Ideal.Quotient.mk I) J

def backward : Ideal (R ⧸ I) → {J : Ideal R // I ≤ J} :=
  fun K => ⟨Ideal.comap (Ideal.Quotient.mk I) K, by
    intro r hr
    rw [comap_mem]
    rw [Ideal.Quotient.eq_zero_iff_mem.mpr hr]
    exact Ideal.zero_mem K⟩

-- Forward-backward composition
lemma forward_backward (K : Ideal (R ⧸ I)) :
    forward I (backward I K) = K := by
  ext x
  constructor
  · intro h
    rw [map_mem] at h
    obtain ⟨r, hr, rfl⟩ := h
    rw [comap_mem] at hr
    exact hr
  · intro hx
    rw [map_mem]
    obtain ⟨r, rfl⟩ := Ideal.Quotient.mk_surjective x
    exact ⟨r, by rwa [comap_mem], rfl⟩

-- Backward-forward composition
lemma backward_forward (J : {J : Ideal R // I ≤ J}) :
    (backward I (forward I J)).val = J.val := by
  ext r
  constructor
  · intro hr
    rw [comap_mem, map_mem] at hr
    obtain ⟨s, hs, hrs⟩ := hr
    -- hrs: mk I s = mk I r, so s - r ∈ I
    have h_diff : s - r ∈ I := by
      rwa [← Ideal.Quotient.mk_eq_mk_iff_sub_mem] at hrs
    -- r = s + (r - s) with s ∈ J, r - s ∈ I ≤ J
    have : r = s + (r - s) := by ring
    rw [this]
    exact Ideal.add_mem _ hs (J.property (neg_mem _ h_diff))
  · intro hr
    rw [comap_mem, map_mem]
    exact ⟨r, hr, rfl⟩

-- Main bijection theorem
theorem correspondence_bijection :
    ∃ φ : {J : Ideal R // I ≤ J} ≃ Ideal (R ⧸ I),
      ∀ J, φ J = Ideal.map (Ideal.Quotient.mk I) J.val := by
  use {
    toFun := forward I
    invFun := backward I
    left_inv := backward_forward I
    right_inv := forward_backward I
  }
  intro; rfl

end BasicCorrespondence

## Part III: Prime Ideal Correspondence (Conservative Approach)

section PrimeCorrespondence

variable {R : Type*} [CommRing R] (I : Ideal R)

-- Comap preserves primality (this should exist)
lemma prime_comap (P : Ideal (R ⧸ I)) [P.IsPrime] :
    (Ideal.comap (Ideal.Quotient.mk I) P).IsPrime := 
  Ideal.IsPrime.comap (Ideal.Quotient.mk I)

-- Forward direction requires more work without advanced API
lemma prime_forward_manual (J : Ideal R) [J.IsPrime] (h : I ≤ J) :
    (Ideal.map (Ideal.Quotient.mk I) J).IsPrime := by
  constructor
  · -- ≠ ⊤
    intro h_top
    have : J = ⊤ := by
      ext r
      constructor
      · intro hr
        have : (Ideal.Quotient.mk I) r ∈ Ideal.map (Ideal.Quotient.mk I) J := by
          rw [map_mem]; exact ⟨r, hr, rfl⟩
        rw [h_top] at this
        trivial
      · intro; trivial
    exact J.isPrime.ne_top this
  · -- Prime multiplication property
    intro a b hab
    obtain ⟨r, rfl⟩ := Ideal.Quotient.mk_surjective a
    obtain ⟨s, rfl⟩ := Ideal.Quotient.mk_surjective b
    rw [← map_mul, map_mem] at hab
    obtain ⟨t, ht, h_eq⟩ := hab
    -- (mk I) t = (mk I) (r * s), so t - r * s ∈ I
    have : t - r * s ∈ I := by
      rwa [← Ideal.Quotient.mk_eq_mk_iff_sub_mem] at h_eq
    -- Since I ≤ J and t ∈ J, we get r * s ∈ J
    have hrs_in_J : r * s ∈ J := by
      have : t = r * s + (t - r * s) := by ring
      rw [← this]
      exact Ideal.add_mem _ ht (h this)
    cases' J.isPrime.mem_or_mem hrs_in_J with hr hs
    · left; rw [map_mem]; exact ⟨r, hr, rfl⟩
    · right; rw [map_mem]; exact ⟨s, hs, rfl⟩

-- Complete prime correspondence
theorem prime_correspondence (J : Ideal R) (h : I ≤ J) :
    J.IsPrime ↔ (Ideal.map (Ideal.Quotient.mk I) J).IsPrime := by
  constructor
  · intro h_prime; exact prime_forward_manual I J h
  · intro h_map_prime
    -- Use correspondence bijection
    obtain ⟨φ, hφ⟩ := correspondence_bijection I
    have : J = (φ.symm (φ ⟨J, h⟩)).val := by simp
    rw [this, hφ]
    exact prime_comap I (φ ⟨J, h⟩)

end PrimeCorrespondence

/-!
## Part IV: Final Noether Theorem
-/

theorem noether_correspondence_theorem (I : Ideal R) :
    ∃ φ : {J : Ideal R // I ≤ J} ≃ Ideal (R ⧸ I),
      (∀ J, φ J = Ideal.map (Ideal.Quotient.mk I) J.val) ∧
      (∀ J, J.val.IsPrime ↔ (φ J).IsPrime) := by
  obtain ⟨φ, hφ⟩ := correspondence_bijection I
  exact ⟨φ, hφ, fun J => prime_correspondence I J.val J.property⟩
