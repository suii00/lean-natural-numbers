-- 🎯 解決策4: 連続関数の非零性保存（核心定理）
theorem continuousAt_ne_of_ne {f : ℝ → ℝ} {t : ℝ} {v : ℝ} 
  (h_cont : ContinuousAt f t) (h_ne : f t ≠ v) : 
  ∃ δ > 0, ∀ s ∈ Set.Ioo (t - δ) (t + δ), f s ≠ v := by
  -- 連続性の定義から直接導出
  have h_open : IsOpen {x | f x ≠ v} := isOpen_ne_fun
  rw [isOpen_iff_mem_nhds] at h_open
  have h_mem : t ∈ {x | f x ≠ v} := h_ne
  have h_nhds : {x | f x ≠ v} ∈ 𝓝 t := h_open t h_mem
  rw [mem_nhds_iff] at h_nhds
  obtain ⟨U, hU_sub, hU_open, ht_mem⟩ := h_nhds
  obtain ⟨δ, hδ_pos, hδ_sub⟩ := Metric.isOpen_iff.mp hU_open t ht_mem
  use δ, hδ_pos
  intros s hs
  exact hU_sub (hδ_sub (Set.mem_of_mem_Ioo hs))

  ∃ δ > 0, ∀ s ∈ Set.Ioo (t - δ) (t + δ), deriv f s ≠ 0 :=
